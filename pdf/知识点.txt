三、temperature温度
1、temperature可以控制AI输出内容的随机性和创造性，值范围0-2之间
2、特点
数值越低，回答越保守、确定，倾向于采用最稳妥、常见的词语回答你
数值越高，回答更开放、随机，也愿意使用一些不常用的词语进行组合来回答你
3、示例代码
from langchain_community.chat_models.tongyi import ChatTongyi
from dotenv import load_dotenv
load_dotenv()
llm = ChatTongyi(model='qwen3-max', temperature=0)
result = llm.stream('请用一种非常独特的、非人类的视角，给‘爱情’下一个定义。')
for chunk in result:
print(chunk.content, end='')
print('\n')
print('-' * 30)

from langchain.chains import ConversationChain
from langchain.memory import ConversationBufferMemory
load_dotenv()
llm = ChatTongyi(model='qwen3-max')
# ConversationBufferMemory() 是专门用来给模型添加记忆的（之前需要一个聊天列表需要每次对话把聊天列表传递
给模型）
# return_messages=True 作用是为了让咱们查看记忆相关的内容
emeory = ConversationBufferMemory(return_messages=True)
# ConversationChain 链，是langchain里面一个重要的模块，把模型和其他模块进行串联
chain = ConversationChain(llm=llm, memory=emeory)

result1 = chain.invoke('我喜欢吃苹果')
print(result1['response'])
print('-' * 20)
result2 = chain.invoke('我喜欢吃什么水果')
print(result2['response'])
print('-' * 20)
# emeory.load_memory_variables({}) 查看记忆相关内容
# history 历史记录
print(emeory.load_memory_variables({}))
