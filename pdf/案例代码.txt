三、temperature温度
1、temperature可以控制AI输出内容的随机性和创造性，值范围0-2之间
2、特点
数值越低，回答越保守、确定，倾向于采用最稳妥、常见的词语回答你
数值越高，回答更开放、随机，也愿意使用一些不常用的词语进行组合来回答你
3、示例代码
from langchain_community.chat_models.tongyi import ChatTongyi
from dotenv import load_dotenv
load_dotenv()
llm = ChatTongyi(model='qwen3-max', temperature=0)
result = llm.stream('请用一种非常独特的、非人类的视角，给‘爱情’下一个定义。')
for chunk in result:
print(chunk.content, end='')
print('\n')
print('-' * 30)

2、为什么要学习Prompt？

from sys import prefix

from langchain_community.chat_models.tongyi import ChatTongyi
from dotenv import load_dotenv
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder, PromptTemplate,
FewShotPromptTemplate
load_dotenv()
llm = ChatTongyi(model='qwen3-max')
# 准备的示例
examples = [
{"comment": "这绝对是今年我用过最棒的耳机，音质清晰，降噪无敌！", "sentiment": "正面"},
{"comment": "太差劲了，才用了一周就坏了，客服也联系不上。", "sentiment": "负面"},
{"comment": "包裹已经收到了，外包装有点磨损。", "sentiment": "中立"}
]
examples_template = PromptTemplate.from_template(
'评论：{comment}\n情感：{sentiment}'
)
# examples 小样本（例子）
# example_prompt 最终以我模版里面的格式输出
# prefix 系统指令（system）
# suffix 用户输入（用户提示语）
# 小样本提示的作用，主要是为了通过例子来引导AI给出标准的回复
few_shot_prompt = FewShotPromptTemplate(
examples=examples,
example_prompt=examples_template,
prefix='请根据以下示例，判断评论的情感（正面、负面、中立）',
suffix='评论：{comment}\n情感',
input_variables=['comment'],
)
final_prompt = few_shot_prompt.format(comment='手机收到了，系统非常流畅')
result = llm.invoke(final_prompt)
print(result.content)
